**注:必须要有对应Controller实现类,swager才能生成对应接口**

## 01 Controller接口类

```java
package com.saddyfire.controller_api;

import com.saddyfire.common.PageResult;
import com.saddyfire.common.Result;
import com.saddyfire.pojo.Agent;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import org.springframework.web.bind.annotation.*;

/**
 * @author SaddyFire
 * @date 2021/12/8
 * @TIME:20:16
 */

@Api(description = "经纪人管理")
public interface AgentManageControllerApi {

    @ApiOperation("根据条件经纪人管理页面的回显")
    @GetMapping
    PageResult page(Integer page,Integer pageSize,@RequestBody Agent agent );

    @ApiOperation("审核经纪人")
    @PostMapping("/check")
    Result check(@RequestBody Agent agent);


}
```
